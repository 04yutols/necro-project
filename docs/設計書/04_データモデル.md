## 1. Prisma スキーマ（最新版）

```prisma
// prisma/schema.prisma

model User {
  id              String      @id @default(uuid())
  name            String
  characters      Character[]
  discoveredItems Item[]      @relation("Discoverer")
}

model Character {
  id              String   @id @default(uuid())
  name            String
  userId          String?
  user            User?    @relation(fields: [userId], references: [id])

  // 9種の基礎ステータス (GDD-003)
  hp, mp, atk, def, matk, mdef, agi, luck, tec: Int

  // 永続パッシブ補正 (GDD-004) — 転職後もリセットされない
  passiveAtkBonus, passiveDefBonus, passiveMatkBonus, passiveMdefBonus: Int @default(0)

  // 職業システム (GDD-004)
  currentJobId    String?
  jobs            UserJob[]

  // 8枠装備スロット (GDD-007)
  equipWeaponId, equipSubId, equipHeadId, equipBodyId,
  equipArmsId, equipLegsId, equipAcc1Id, equipAcc2Id: String?
  // → Item モデルへの @relation (8本)

  // 死霊術ステータス (GDD-005)
  necroLevel          Int   @default(1)   // 1〜99
  necroRank           Int   @default(1)   // 1〜10
  necroMaxCost        Int   @default(10)  // パーティ最大コスト
  necroBaseStatsBonus Float @default(1.0) // ランクアップ倍率

  // 進行管理
  clearedStages   String[] @default([])

  // 深淵の残滓
  abyssalResidues AbyssalResidue[]
}

model Job {
  id       String    @id   // "warrior" など
  name     String
  tier     Int             // 1 or 2
  category String          // "PHYSICAL" | "MAGICAL"
  userJobs UserJob[]
}

// 職業ごとの独立レベル管理 — 転職してもこのレコードは残る (GDD-004)
model UserJob {
  characterId String
  jobId       String
  level       Int    @default(1)
  exp         Int    @default(0)
  character   Character @relation(fields: [characterId], references: [id])
  job         Job       @relation(fields: [jobId], references: [id])
  @@id([characterId, jobId])
}

model Item {
  id            String   @id @default(uuid())
  name          String
  type          String   // "WEAPON"|"SUB"|"HEAD"|"BODY"|"ARMS"|"LEGS"|"ACC1"|"ACC2"
  rarity        String   // "COMMON"|"UNIQUE"|"HIDDEN_UNIQUE"
  atk, def, matk, mdef: Int @default(0)
  specialEffect String?

  // 第一発見者システム (GDD-追加要件)
  isUnique        Boolean   @default(false)
  discovererId    String?
  discovererName  String?
  discoverer      User?     @relation("Discoverer", fields: [discovererId], references: [id])
  serialNo        Int?      // グローバル発見順シーケンス
  discoveredAt    DateTime?

  // ランダムオプション (JSONB)
  subOptions      Json?     // Array<{ type: string, value: number }>

  residues        AbyssalResidue[]
  // + 8本の equippedXxx リレーション（逆参照）
}

model Monster {
  id       String  @id @default(uuid())
  name     String
  tribe    String  // "UNDEAD"|"DEMON"|"BEAST"|"HUMANOID"
  cost     Int
  // 9種ステータス（フラットカラム）
  hp, mp, atk, def, matk, mdef, agi, luck, tec: Int

  soulShardId  String?
  soulShard    SoulShard?  @relation(fields: [soulShardId], references: [id])
  spiritCoreId String?
  spiritCore   SpiritCore? @relation(fields: [spiritCoreId], references: [id])
}

model SoulShard {
  id              String    @id @default(uuid())
  originMonster   String    // 変換元モンスター名
  atkBonus        Int       @default(0)
  matkBonus       Int       @default(0)
  specialAbility  String?   // "REGENERATE_SOUL"|"BANE_OF_LIGHT" など
  monsters        Monster[]
}

model SpiritCore {
  id           String    @id @default(uuid())
  name         String
  element      String?   // ElementType
  skillChangeId String?
  atkMultiplier Float    @default(1.0)
  monsters     Monster[]
}

model AbyssalResidue {
  id          String     @id @default(uuid())
  name        String
  itemId      String
  item        Item       @relation(fields: [itemId], references: [id])
  characterId String?
  character   Character? @relation(fields: [characterId], references: [id])
  subOptions  Json       // Array<{ type: string, value: number }>
  level       Int        @default(1)  // 1〜20
}
```

## 2. TypeScript 型定義（src/types/game.ts 全リスト）

```typescript
// ユニオン型
type ClassCategory = 'PHYSICAL' | 'MAGICAL'

interface SubOption { type: string; value: number }

  weapon,sub,head,body,arms,legs,acc1,acc2: ItemData | null
}

interface CharacterData {
  id: string; name: string; currentJobId: string; category: ClassCategory;
  stats: BaseStats; passives: PassiveBonuses; equipment: EquipmentSlots;
  baseResistances: Resistances; jobs: UserJobState[];
  isAwakened: boolean; clearedStages: string[];
}

// アイテム
interface ItemData {

  specialEffect?: string; isUnique: boolean;

  subOptions?: SubOption[];

// モンスター・死霊術
```

## 4. ステータス最終計算式

```
finalStats = BaseStats + PassiveBonuses + Equipment合算
```

1. `BaseStats`: キャラクターの9ステータス（DBの `Character.hp` 等）
2. `PassiveBonuses`: 職業レベルマイルストーンで累積（`passiveAtkBonus` 等）
3. `Equipment`: 8スロットの `ItemData.stats` を全てキーごとに加算

```typescript
// BattleEngine.getTotalStats() の実装
const total = { ...player.stats };
total.atk  += player.passives.passiveAtkBonus;
total.def  += player.passives.passiveDefBonus;
total.matk += player.passives.passiveMatkBonus;
total.mdef += player.passives.passiveMdefBonus;
// ※ 現時点: 装備ステータスは BattleEngine に渡されていない（要将来対応）
```

## 5. 報酬計算アルゴリズム

### EXP計算（RewardService.calculateExp）
```
gainedExp = baseExp × (1 + jobLevel/100) × categoryMultiplier
categoryMultiplier: MAGICAL=1.1 / PHYSICAL=1.0
```

### ドロップ率計算（RewardService.calculateDrops）
```
adjustedRate = baseRate × (1 + LUCK/100)
// LUCK=50 → 1.5倍、LUCK=100 → 2倍
drop = Math.random() < adjustedRate
```

## 6. 深淵の残滓（AbyssalResidue）成長式

```typescript
// upgradeResidue() 内のロジック
let newExp = residue.exp + gainedExp;
let newLevel = residue.level;
let newMaxExp = residue.maxExp;

while (newExp >= newMaxExp && newLevel < 20) {
  newExp -= newMaxExp;
  newLevel++;
  newMaxExp = Math.floor(newMaxExp * 1.5);  // 1.5倍ずつ必要EXP増加
}
```

初期 maxExp の例: Lv1=800, Lv2=1200, Lv3=1800, ... Lv20=上限

## 7. 第一発見者システム DB 設計

```
HIDDEN_UNIQUE アイテムがドロップ
  → PostgreSQL グローバルシーケンスで serialNo を発番
  → Item.isUnique = true
  → Item.discovererId = User.id
  → Item.discovererName = User.name
  → Item.discoveredAt = 現在時刻
  → 以降そのアイテムを取得した全プレイヤーに discovererName が表示される
```

**設計意図**: `serialNo` はグローバル採番なので「世界で何番目に見つけた」が分かる。`isUnique=true` かつ `discovererId` が存在する場合のみ鑑定証明書の生成が可能。
