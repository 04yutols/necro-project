## 3. DB連携バトル結合

### 3-a. フロー全体

```
BattleCanvas (client)
  ↓ 全WAVEクリア
  processStageResultAction(stageId)  ← Server Action
  ↓
  auth() でセッション取得
  ↓ 未ログイン
  RewardService.processDropTable (client-side drop roll)  ← ゲスト動作
  ↓ ログイン済み
  GameManager.processStageResult(characterId, stageId)
    → RewardService.processDropTable (server-side)
    → prisma.$transaction:
        - Item.create × weapons
        - AbyssalResidue.create × residues
        - Character.update (clearedStages push)
        - UserJob.update (exp, level)
    → return StageDropResult + expGain + goldGain
  ↓
BattleCanvas
  addInventoryItems / addAbyssalResidues / addResidueMaterials
  addClearedStage / addExp / addGold
  setBattleResult({ itemsGained: convertToResultDrops(dropResult) })
  setShowResult(true)
```

### 3-b. `src/app/actions.ts` — processStageResultAction（実装）

```typescript
'use server';

import { auth } from '@/auth';
import { prisma } from '@/lib/prisma';
import { RewardService, StageDropResult } from '@/services/RewardService';
import { MasterDataService } from '@/services/MasterDataService';
import type { ItemData, AbyssalResidueData, ResidueMatData } from '@/types/game';

export interface StageResultPayload {
  success: boolean;
  dropResult: StageDropResult;
  expGain: number;
  goldGain: number;
  error?: string;
}

export async function processStageResultAction(stageId: string): Promise<StageResultPayload> {
  const mds    = MasterDataService.getInstance();
  const stage  = mds.getStage(stageId);
  const svc    = new RewardService();

  // ゲスト動作: 未ログインはクライアント側で drop roll のみ
  const session = await auth();
  if (!session?.user?.id) {
    const dropResult = svc.processDropTable(stage?.rewards.dropTable ?? []);
    return {
      success: true,
      dropResult,
      expGain:  stage?.rewards.baseExp ?? 0,
      goldGain: stage?.rewards.baseGold ?? 0,
    };
  }

  if (!stage) return { success: false, dropResult: emptyDrop(), expGain: 0, goldGain: 0, error: 'Stage not found' };

  // Character 取得
  const char = await prisma.character.findFirst({
    where: { userId: session.user.id },
    include: { jobs: true },
  });
  if (!char) return { success: false, dropResult: emptyDrop(), expGain: 0, goldGain: 0, error: 'Character not found' };

  // EXP計算
  const playerData: any = { category: 'PHYSICAL', currentJobId: char.currentJobId, jobs: char.jobs };
  const expGain  = svc.calculateExp(stage.rewards.baseExp, playerData);
  const goldGain = stage.rewards.baseGold;

  // ドロップ抽選（サーバー側）
  const dropResult = svc.processDropTable(stage.rewards.dropTable);

  // DB保存 (トランザクション)
  await prisma.$transaction(async (tx) => {
    // 武器保存
    for (const weapon of dropResult.weapons) {
      await tx.item.create({
        data: {
          name:      weapon.name,
          type:      weapon.type,
          rarity:    weapon.rarity,
          rank:      weapon.rank ?? 0,
          archetype: weapon.archetype ?? null,
          ilv:       weapon.ilv ?? null,
          passiveA:  weapon.passiveA ? (weapon.passiveA as any) : undefined,
          passiveB:  weapon.passiveB ? (weapon.passiveB as any) : undefined,
          subOptions: weapon.subOptions ? (weapon.subOptions as any) : undefined,
          isUnique:  weapon.isUnique,
        },
      });
    }

    // 残滓保存
    for (const residue of dropResult.residues) {
      await tx.abyssalResidue.create({
        data: {
          name:        residue.name,
          itemId:      residue.itemId,
          characterId: char.id,
          rarity:      residue.rarity,
          mainStat:    residue.mainStat as any,
          subOptions:  residue.subOptions as any,
          level:       residue.level,
          exp:         residue.exp,
          maxExp:      residue.maxExp,
        },
      });
    }

    // EXP加算
    const currentJob = char.jobs.find((j: any) => j.jobId === char.currentJobId);
    if (currentJob) {
      const newExp   = currentJob.exp + expGain;
      const newLevel = Math.min(99, Math.floor(newExp / 100) + 1);
      await tx.userJob.update({
        where: { characterId_jobId: { characterId: char.id, jobId: char.currentJobId! } },
        data:  { exp: newExp, level: newLevel },
      });
    }

    // クリアフラグ
    if (!char.clearedStages.includes(stageId)) {
      await tx.character.update({
        where: { id: char.id },
        data:  { clearedStages: { push: stageId } },
      });
    }
  });

  return { success: true, dropResult, expGain, goldGain };
}

function emptyDrop(): StageDropResult {
  return { weapons: [], residues: [], materials: [], monsters: [] };
}
```

**変換ヘルパー（BattleCanvas 内に追加）:**

```typescript
function convertDropToResultItems(drop: StageDropResult, playerName?: string) {
  const weapons = drop.weapons.map(w => ({
    id:       w.id,
    name:     w.name,
    type:     w.type as any,
    rarity:   (w.rarity === 'R' ? 'RARE' : w.rarity) as any,
    icon:     '⚔',
    isUnique: w.isUnique,
    flavor:   w.flavor,
  }));
  const residues = drop.residues.map(r => ({
    id:     r.id,
    name:   r.name,
    rarity: (r.rarity === 'RARE' ? 'SR' : r.rarity === 'EPIC' ? 'SSR' : 'COMMON') as any,
    icon:   '◆',
    isUnique: false,
  }));
  const materials = drop.materials.map(m => ({
    id:       m.id,
    name:     m.name,
    rarity:   (m.rarity === 'RARE' ? 'SR' : 'COMMON') as any,
    icon:     '▣',
    isUnique: false,
    quantity: m.quantity,
  }));
  return [...weapons, ...residues, ...materials];
}
```
