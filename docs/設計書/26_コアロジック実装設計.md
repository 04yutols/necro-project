### 1.4 `selectEnemyTarget()` — 新規メソッド

生存モンスターから隊列ヘイト重みで重み付き抽選する。

```typescript
private selectEnemyTarget(): MonsterData | null {
  const HATE_WEIGHTS = [50, 30, 20] as const;
  const { monsters } = this.state;

  const aliveMonsters = monsters
    .map((m, idx) => m ? { monster: m, idx } : null)
    .filter((x): x is { monster: MonsterData; idx: number } =>
      x !== null && (this.monsterCurrentHp[x.monster.id] ?? 0) > 0
    );

  if (aliveMonsters.length === 0) return null;  // 全滅 → アルドが直接受ける

  const totalWeight = aliveMonsters.reduce(
    (sum, { idx }) => sum + (HATE_WEIGHTS[idx] ?? 20), 0
  );
  let rand = Math.random() * totalWeight;
  for (const { monster, idx } of aliveMonsters) {
    rand -= HATE_WEIGHTS[idx] ?? 20;
    if (rand <= 0) return monster;
  }
  return aliveMonsters[aliveMonsters.length - 1].monster;
}
```

```typescript
private processEnemyCounterAttack(enemy: MonsterData): void {
  const { player } = this.state;
  const target = this.selectEnemyTarget();

  if (target) {
    // モンスターへの攻撃
    const rawDmg = Math.max(1, Math.floor(
      enemy.stats.atk * (1 - target.stats.def / (target.stats.def + 200))
    ));
    // ORC シナジー: absorbDmgPct 分を HP 吸収
    const sb = this.synergyBonus;
    const absorbed = sb.absorbDmgPct
      ? Math.floor(rawDmg * sb.absorbDmgPct / 100)
      : 0;
    const finalDmg = Math.max(1, rawDmg - absorbed);

    this.monsterCurrentHp[target.id] = Math.max(
      0,
      (this.monsterCurrentHp[target.id] ?? target.stats.hp) - finalDmg
    );

    const isDead = this.monsterCurrentHp[target.id] <= 0;
    const desc = isDead
      ? `${enemy.name}の攻撃！ ${target.name}は倒れた！`
      : `${enemy.name}の攻撃！ ${target.name}に${finalDmg}ダメージ${absorbed > 0 ? `（${absorbed}吸収）` : ''}。`;

    this.addLog('ENEMY_ATTACK', enemy.name, target.name, desc, finalDmg, false, false, false, 'NONE', 'STRIKE');
  } else {
    // モンスター全滅 → アルドが直接受ける
    const ms = this.getMutableStats(player);
    const rawDmg = Math.max(1, Math.floor(
      enemy.stats.atk * (1 - ms.def / (ms.def + 200))
    ));
    ms.hp = Math.max(1, ms.hp - rawDmg);
    this.addLog('ENEMY_ATTACK', enemy.name, player.name, `${enemy.name}の攻撃！ アルドに${rawDmg}ダメージ。`, rawDmg, false, false, false, 'NONE', 'STRIKE');
  }
}
```

### 2.2 ギミック仕様

| trigger | 条件 | effect | 動作 |
|---|---|---|---|
| `HP_BELOW_50` | ボスHP が 50% 以下に下がった瞬間（1回限り） | `ENRAGE` | ボス ATK × 1.5、次ターン予告ログ |
| `TURN_3` | ターン3開始時（1回限り） | `AV_DELAY` | プレイヤーの行動を `value` ターン遅延（PARALYSIS相当） |
| `ON_SHIELD_BREAK` | 霊的防壁が破壊された瞬間 | `SUMMON_MINIONS` | MINION を 1〜2体追加（次WAVE扱い） |
| `ON_REVIVE` | `REVIVE` 後のギミックリセット用フック | — | firedGimmicks をクリアしてギミックを再度有効化 |
| `HP_BELOW_50` | HP 0 → `REVIVE` が設定されている場合 | `REVIVE` | HP を 50% に回復し `shieldBroken = false` にリセット（1回限り） |

### 2.3 `checkBossGimmicks()` — 新規メソッド

```typescript
private checkBossGimmicks(
  boss: MonsterData,
  prevHpPct: number,
  newHpPct: number
): void {
  if (!boss.gimmicks) return;

  for (const g of boss.gimmicks) {
    const key = `${boss.id}:${g.trigger}:${g.effect}`;
    if (this.firedGimmicks.has(key)) continue;

    const shouldFire = (() => {
      switch (g.trigger) {
        case 'HP_BELOW_50':
          return prevHpPct > 50 && newHpPct <= 50;
        case 'TURN_3':
          return this.state.turn === 3;
        case 'ON_SHIELD_BREAK':
          return boss.shieldBroken === true;
        case 'ON_REVIVE':
          return false;  // REVIVE 後に手動トリガー
        default:
          return false;
      }
    })();

    if (!shouldFire) continue;
    this.firedGimmicks.add(key);
    this.applyBossGimmickEffect(boss, g);
  }
}

private applyBossGimmickEffect(boss: MonsterData, g: BossGimmick): void {
  const { player } = this.state;

  switch (g.effect) {
    case 'ENRAGE':
      // ATK を永続的に 1.5 倍（ランタイム変更）
      boss.stats = { ...boss.stats, atk: Math.floor(boss.stats.atk * 1.5) };
      this.addLog('BOSS_ENRAGE', boss.name, boss.name,
        `【ENRAGE】${boss.name}が激怒した！ 攻撃力が大幅に上昇する！`);
      break;

    case 'AV_DELAY': {
      // PARALYSIS 相当の行動遅延を付与
      const turns = g.value ?? 1;
      const delayedEffect: StatusEffect = {
        type: 'PARALYSIS',
        remainingTurns: turns,
        stackCount: 1,
      };
      player.statusEffects = [...(player.statusEffects ?? []), delayedEffect];
      this.addLog('BOSS_AV_DELAY', boss.name, player.name,
        `【AV遅延】${boss.name}の特殊攻撃！ アルドの行動が${turns}ターン遅延する！`);
      break;
    }

    case 'REVIVE':
      // HP 50% で復活、防壁リセット
      boss.stats = { ...boss.stats, hp: Math.floor(boss.stats.hp * 0.5) };
      boss.shieldBroken = false;
      boss.shieldHp = boss.maxShieldHp ?? 0;
      // 発火済みギミックをリセット（ON_REVIVE フック）
      for (const g2 of boss.gimmicks ?? []) {
        const k = `${boss.id}:${g2.trigger}:${g2.effect}`;
        if (g2.trigger === 'ON_REVIVE') this.firedGimmicks.delete(k);
      }
      this.addLog('BOSS_REVIVE', boss.name, boss.name,
        `【REVIVE】${boss.name}が第2形態に移行した！ HPが回復し、防壁が再生する！`);
      break;

    case 'SUMMON_MINIONS':
      // BattleLog にフラグを立て、BattleCanvas 側で敵追加演出を行う
      this.addLog('BOSS_SUMMON', boss.name, 'FIELD',
        `【召喚】${boss.name}が手下を呼んだ！`);
      // 実際の敵追加は BattleCanvas 側で battleLogs を監視して処理
      break;
  }
}
```

### 3.5 `processUltimateSkill()` — 魔神技実行（新規）

```typescript
public simulateUltimateSkill(target: MonsterData, enemy: MonsterData): BattleLog[] {
  this.logs = [];
  const demon = this.demonState;
  if (!demon?.isDemonMode || !demon.form || demon.ultimateUsed) return this.logs;

  const { player } = this.state;
  const ult = demon.form.ultimateSkill;
  const stats = calculateCharacterStatProfile(player).total;

  // 防御無視フラグ
  const ignoreDef = shouldBypassDefense(demon.form);
  const defStats = ignoreDef ? { ...target.stats, def: 0 } : target.stats;

  const isAoe = ult.damage.targetType === 'ALL';
  const targets: MonsterData[] = isAoe
    ? (this.state.monsters.filter(Boolean) as MonsterData[])
    : [target];

  for (const t of targets) {
    const { damage, isCritical, isWeakness, isResisted } = this.calculateDamage(
      stats,
      player.elementDmgBoosts ?? {},
      defStats,
      ult.damage.flags?.includes('IGNORE_RESISTANCE') ? {} : (t.resistances ?? {}),
      ult.damage.power,
      ult.damage.element,
    );
    const shieldResult = this.applySpiritualShield(t, damage, ult.damage.element);
    this.addLog('DEMON_ULTIMATE', player.name, t.name,
      `【魔神技】${demon.form.formName}の奥義「${ult.nameJa}」！`,
      shieldResult.damage, isCritical, isWeakness, isResisted,
      ult.damage.element, 'MAGIC');
  }

  // 残留効果ログ（BattleCanvas 側で効果を適用）
  this.addLog('DEMON_LINGERING', player.name, 'FIELD',
    `【残留】${ult.lingering.descJa}（${ult.lingering.duration === -1 ? 'バトル終了まで' : `${ult.lingering.duration}ターン`}）`);

  // 魔神技使用済みフラグ
  this.demonState = markDemonUltimateUsed(demon);

  // 敵の反撃
  this.processEnemyCounterAttack(enemy);
  this.updateState();
  return this.logs;
}
```

```typescript
function evalSoulShatter(passive: WeaponPassive, ctx: WeaponPassiveContext): WeaponPassiveResult | null {
```

```typescript
function evalDemonMode(passive: WeaponPassive, ctx: WeaponPassiveContext): WeaponPassiveResult | null {
  if (ctx.trigger !== 'ON_ATTACK') return null;

  const rank = (ctx.actor.equipment?.weapon as any)?.rank ?? 0;
  const baseVal = passive.values[0] ?? 5;  // ゲージ増加量（例: 5%）
  const maxVal  = passive.values[1] ?? 10;
  const gaugeDelta = Math.floor(baseVal + (maxVal - baseVal) * (rank / 5));

  // condition が "DEMON_ACTIVE" の場合は魔神化中のみ追加ダメージとして機能
  if (passive.condition === 'DEMON_ACTIVE') {
    if (!ctx.isDemonMode) return null;
    // ゲージ増加の代わりに追加ダメージ倍率として返す
    const dmgBonus = Math.floor(ctx.actor.stats.atk * (gaugeDelta / 100));
    return {
      bonusDamage: dmgBonus,
      logDesc: `【魔神昂揚】${passive.nameJa}が発動！ 追加 ${dmgBonus} ダメージ。`,
    };
  }

  return {
    demonGaugeDelta: gaugeDelta,
    logDesc: `【魂の高揚】${passive.nameJa}が発動！ 魔神化ゲージ +${gaugeDelta}。`,
  };
}
```

### 4.4 メインエクスポート関数

```typescript
export function evaluateWeaponPassive(
  passive: WeaponPassive | undefined,
  ctx: WeaponPassiveContext,
): WeaponPassiveResult | null {
  if (!passive) return null;

  switch (passive.systemTag) {
    case 'SOUL_SHATTER':   return evalSoulShatter(passive, ctx);
    case 'ACTION_VALUE':   return evalActionValue(passive, ctx);
    case 'DEMON_MODE':     return evalDemonMode(passive, ctx);
    case 'SHIELD_PIERCE':  return null;  // 第2章以降
    case 'GIANT_KILLING':  return null;  // 第2章以降
    default:               return null;
  }
}
```

### 4.5 BattleEngine への組み込みポイント

```typescript
// processPlayerAction 内、ダメージ計算後に追加

const weapon = player.equipment?.weapon;
if (weapon) {
  for (const passive of [weapon.passiveA, weapon.passiveB]) {
    // ON_ATTACK 判定
    const attackCtx: WeaponPassiveContext = {
      trigger: 'ON_ATTACK',
      actor: player,
      target,
      isCritical,
      didBreakShield: shieldResult.didBreak,
      isDemonMode: this.demonState?.isDemonMode ?? false,
    };
    const attackResult = evaluateWeaponPassive(passive, attackCtx);
    if (attackResult) this.applyPassiveResult(attackResult, player, target);

    // ON_SHIELD_BREAK 判定（防壁破壊時のみ）
    if (shieldResult.didBreak) {
      const shieldCtx = { ...attackCtx, trigger: 'ON_SHIELD_BREAK' as const };
      const shieldResult2 = evaluateWeaponPassive(passive, shieldCtx);
      if (shieldResult2) this.applyPassiveResult(shieldResult2, player, target);
    }
  }
}
```

```typescript
private applyPassiveResult(
  result: WeaponPassiveResult,
  player: CharacterData,
  target: MonsterData,
): void {
  if (result.bonusDamage) {
    target.stats.hp = Math.max(0, target.stats.hp - result.bonusDamage);
  }
  if (result.demonGaugeDelta) {
    // BattleLog 経由で useGameStore.fillDemonGauge() をトリガー
    this.addLog('PASSIVE_DEMON_GAUGE', player.name, player.name,
      result.logDesc ?? '', undefined, false, false, false, 'NONE', 'MAGIC');
  }
  if (result.avReduction) {
    this.addLog('PASSIVE_AV_BOOST', player.name, player.name,
      result.logDesc ?? '', undefined, false, false, false, 'NONE', 'MAGIC');
  }
  if (result.logDesc && !result.demonGaugeDelta && !result.avReduction) {
    this.addLog('PASSIVE_TRIGGER', player.name, target.name,
      result.logDesc, result.bonusDamage, false, false, false, 'NONE', 'MAGIC');
  }
}
```
